int g(int x){
	int res;
	int j;
	int k;
	res = 0;
        for (j=0; j< 2; j=j+1){
		if (j == 1){
			break;
		} else{
			for (k=0; k< x; k=k+1){
				res = res + (k + 1);
			}
		} endif
	}
	return res;
}

		output(b[i]);

			break;
		endif

}
